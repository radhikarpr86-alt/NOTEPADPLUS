<!-- ============================================

<script id="helpdesk-ticketing-system">
  // ==========================================
  // KONFIGURASI HELPDESK
  // ==========================================
  const HELPDESK_CONFIG = {
    storagePrefix: APP_CONFIG.storagePrefix + 'helpdesk_',
    priorities: ['low', 'medium', 'high', 'urgent'],
    statuses: ['open', 'in_progress', 'waiting', 'resolved', 'closed'],
    categories: ['Billing', 'Akun', 'Teknis', 'Fitur', 'Ojek', 'Toko', 'Hello Friends', 'Lainnya'],
    slaByPriority: {
      low: { responseHours: 24, resolutionHours: 72 },
      medium: { responseHours: 8, resolutionHours: 24 },
      high: { responseHours: 2, resolutionHours: 8 },
      urgent: { responseHours: 0.5, resolutionHours: 2 }
    },
    maxTicketsPerUser: 20,
    autoCloseAfterDays: 7 // Auto-close ticket setelah 7 hari resolved
  };

  // ==========================================
  // DATA STORE
  // ==========================================

  function getHelpdeskData(key) {
    return getFromStorage(HELPDESK_CONFIG.storagePrefix + key, []);
  }

  function saveHelpdeskData(key, data) {
    return saveToStorage(HELPDESK_CONFIG.storagePrefix + key, data);
  }

  // ==========================================
  // TICKET MANAGER
  // ==========================================

  const TicketManager = {
    /**
     * Buat tiket baru
     */
    createTicket(data) {
      const { userId, userName = '', userRole = '', subject = '', message = '', category = 'Lainnya', priority = 'medium', attachment = '' } = data;
      const tickets = getHelpdeskData('tickets');

      // Cek batas tiket per user
      const userOpenTickets = tickets.filter(t => t.userId === userId && t.status !== 'closed' && t.status !== 'resolved');
      if (userOpenTickets.length >= HELPDESK_CONFIG.maxTicketsPerUser) {
        return { success: false, message: `Maksimal ${HELPDESK_CONFIG.maxTicketsPerUser} tiket terbuka per user` };
      }

      const sla = HELPDESK_CONFIG.slaByPriority[priority] || HELPDESK_CONFIG.slaByPriority.medium;
      const ticket = {
        id: generateId('TKT'),
        ticketNumber: `TKT-${formatDate(new Date(), 'YYYYMMDD')}-${generateRandomString(4).toUpperCase()}`,
        userId, userName, userRole, subject, message, category, priority, attachment,
        status: 'open',
        assignedTo: null,
        assignedName: '',
        responses: [],
        sla: {
          responseDeadline: new Date(Date.now() + sla.responseHours * 3600000).toISOString(),
          resolutionDeadline: new Date(Date.now() + sla.resolutionHours * 3600000).toISOString()
        },
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
        resolvedAt: null,
        closedAt: null
      };

      tickets.push(ticket);
      saveHelpdeskData('tickets', tickets);

      // Notifikasi ke CS
      sendRoleNotification('cs', {
        category: 'system',
        title: '🎫 Tiket Baru',
        message: `[${ticket.ticketNumber}] ${ticket.subject} - ${ticket.category} (${ticket.priority})`,
        actionUrl: 'tiket',
        actionType: 'navigate'
      });

      systemLogger.info('helpdesk', `Ticket created: ${ticket.ticketNumber}`);
      return { success: true, ticket };
    },

    /**
     * Assign tiket ke CS
     */
    assignTicket(ticketId, csId) {
      const tickets = getHelpdeskData('tickets');
      const index = tickets.findIndex(t => t.id === ticketId);
      if (index === -1) return null;

      const cs = getCSById(csId);
      tickets[index].assignedTo = csId;
      tickets[index].assignedName = cs?.nama || 'CS';
      tickets[index].status = tickets[index].status === 'open' ? 'in_progress' : tickets[index].status;
      tickets[index].updatedAt = new Date().toISOString();
      saveHelpdeskData('tickets', tickets);
      return tickets[index];
    },

    /**
     * Tambah respon ke tiket
     */
    addResponse(ticketId, data) {
      const { userId, userName = '', message = '', isStaff = false, attachment = '' } = data;
      const tickets = getHelpdeskData('tickets');
      const index = tickets.findIndex(t => t.id === ticketId);
      if (index === -1) return null;

      const response = {
        id: generateId('RSP'),
        userId, userName, message, isStaff, attachment,
        createdAt: new Date().toISOString()
      };

      tickets[index].responses.push(response);
      tickets[index].updatedAt = new Date().toISOString();

      // Update status
      if (isStaff && tickets[index].status === 'open') {
        tickets[index].status = 'in_progress';
      }
      if (!isStaff && tickets[index].status === 'waiting') {
        tickets[index].status = 'in_progress';
      }

      saveHelpdeskData('tickets', tickets);

      // Notifikasi ke pihak lain
      const ticket = tickets[index];
      const notifyUserId = isStaff ? ticket.userId : (ticket.assignedTo || 'cs');
      sendNotification(notifyUserId, {
        category: 'system',
        title: `💬 Respon Tiket ${ticket.ticketNumber}`,
        message: `${userName}: ${message.substring(0, 100)}`,
        actionUrl: 'tiket',
        actionType: 'navigate'
      });

      return ticket;
    },

    /**
     * Update status tiket
     */
    updateStatus(ticketId, status, note = '') {
      const tickets = getHelpdeskData('tickets');
      const index = tickets.findIndex(t => t.id === ticketId);
      if (index === -1) return null;

      tickets[index].status = status;
      tickets[index].updatedAt = new Date().toISOString();

      if (status === 'resolved') tickets[index].resolvedAt = new Date().toISOString();
      if (status === 'closed') tickets[index].closedAt = new Date().toISOString();

      if (note) {
        tickets[index].responses.push({
          id: generateId('RSP'),
          userId: getCurrentUserId() || 'system',
          userName: 'System',
          message: `Status diubah ke "${status}". ${note}`,
          isStaff: true,
          createdAt: new Date().toISOString()
        });
      }

      saveHelpdeskData('tickets', tickets);

      // Notifikasi ke user
      if (tickets[index].userId) {
        sendNotification(tickets[index].userId, {
          category: 'system',
          title: status === 'resolved' ? '✅ Tiket Diselesaikan' : '📋 Status Tiket Diupdate',
          message: `Tiket ${tickets[index].ticketNumber}: ${status}`,
          actionUrl: 'tiket',
          actionType: 'navigate'
        });
      }

      return tickets[index];
    },

    /**
     * Dapatkan tiket dengan filter
     */
    getTickets(filters = {}) {
      let tickets = getHelpdeskData('tickets');

      if (filters.userId) tickets = tickets.filter(t => t.userId === filters.userId);
      if (filters.status) tickets = tickets.filter(t => t.status === filters.status);
      if (filters.priority) tickets = tickets.filter(t => t.priority === filters.priority);
      if (filters.category) tickets = tickets.filter(t => t.category === filters.category);
      if (filters.assignedTo) tickets = tickets.filter(t => t.assignedTo === filters.assignedTo);
      if (filters.search) {
        const s = filters.search.toLowerCase();
        tickets = tickets.filter(t =>
          t.subject.toLowerCase().includes(s) || t.ticketNumber.toLowerCase().includes(s) || t.message.toLowerCase().includes(s)
        );
      }

      return tickets.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    },

    /**
     * Dapatkan tiket yang melanggar SLA
     */
    getSLABreached() {
      const tickets = getHelpdeskData('tickets').filter(t => t.status === 'open' || t.status === 'in_progress');
      const now = new Date();
      return tickets.filter(t => t.sla?.responseDeadline && new Date(t.sla.responseDeadline) < now);
    },

    /**
     * Dapatkan statistik helpdesk
     */
    getStats() {
      const tickets = getHelpdeskData('tickets');
      const now = new Date();

      const byStatus = {};
      HELPDESK_CONFIG.statuses.forEach(s => { byStatus[s] = tickets.filter(t => t.status === s).length; });

      const byPriority = {};
      HELPDESK_CONFIG.priorities.forEach(p => { byPriority[p] = tickets.filter(t => t.priority === p).length; });

      const byCategory = {};
      HELPDESK_CONFIG.categories.forEach(c => { byCategory[c] = tickets.filter(t => t.category === c).length; });

      const avgResponseTime = this.calculateAvgResponseTime();

      return {
        total: tickets.length,
        open: byStatus['open'] || 0,
        inProgress: byStatus['in_progress'] || 0,
        resolved: byStatus['resolved'] || 0,
        slaBreached: this.getSLABreached().length,
        avgResponseTime,
        byStatus, byPriority, byCategory
      };
    },

    /**
     * Hitung rata-rata response time
     */
    calculateAvgResponseTime() {
      const tickets = getHelpdeskData('tickets').filter(t => t.responses.length > 0);
      if (tickets.length === 0) return 0;

      let totalMinutes = 0;
      let count = 0;

      tickets.forEach(t => {
        const firstResponse = t.responses[0];
        if (firstResponse && firstResponse.isStaff) {
          const diff = (new Date(firstResponse.createdAt) - new Date(t.createdAt)) / 60000;
          totalMinutes += diff;
          count++;
        }
      });

      return count > 0 ? Math.round(totalMinutes / count) : 0;
    },

    /**
     * Auto-close tiket yang sudah resolved > 7 hari
     */
    autoCloseResolved() {
      const tickets = getHelpdeskData('tickets');
      const cutoff = new Date(Date.now() - HELPDESK_CONFIG.autoCloseAfterDays * 86400000);
      let closed = 0;

      tickets.forEach(t => {
        if (t.status === 'resolved' && t.resolvedAt && new Date(t.resolvedAt) < cutoff) {
          t.status = 'closed';
          t.closedAt = new Date().toISOString();
          closed++;
        }
      });

      if (closed > 0) saveHelpdeskData('tickets', tickets);
      return closed;
    }
  };

  /**
   * Render daftar tiket
   */
  function renderTicketsList(tickets) {
    if (tickets.length === 0) return '<p class="text-muted text-center py-4">Tidak ada tiket</p>';

    return tickets.map(ticket => {
      const priorityColors = { urgent: 'badge-danger', high: 'badge-warning', medium: 'badge-info', low: 'badge-neutral' };
      const statusColors = { open: 'badge-danger', in_progress: 'badge-warning', waiting: 'badge-info', resolved: 'badge-success', closed: 'badge-neutral' };

      return `
        <div class="card card-flat mb-2" style="border-left: 4px solid ${ticket.priority === 'urgent' ? 'var(--color-danger)' : ticket.priority === 'high' ? 'var(--color-warning)' : ticket.priority === 'medium' ? 'var(--color-info)' : '#999'};">
          <div class="card-body">
            <div class="flex items-center justify-between flex-wrap gap-3">
              <div class="flex-1 min-w-0">
                <div class="flex items-center gap-2 mb-1 flex-wrap">
                  <span class="badge badge-neutral">${ticket.ticketNumber}</span>
                  <span class="badge ${priorityColors[ticket.priority] || 'badge-neutral'}">${ticket.priority}</span>
                  <span class="badge ${statusColors[ticket.status] || 'badge-neutral'}">${ticket.status.replace(/_/g, ' ')}</span>
                  <span class="badge badge-neutral">${ticket.category}</span>
                </div>
                <strong class="text-sm">${ticket.subject}</strong>
                <div class="text-xs text-muted mt-1">
                  👤 ${ticket.userName} • 🕐 ${formatRelativeDate(ticket.createdAt)} • 💬 ${ticket.responses.length} respon
                  ${ticket.assignedName ? `• 👨‍💼 ${ticket.assignedName}` : '• ⚠️ Belum di-assign'}
                  ${ticket.sla?.responseDeadline && new Date(ticket.sla.responseDeadline) < new Date() ? '• <span class="text-danger">SLA Breached!</span>' : ''}
                </div>
              </div>
              <div class="flex gap-1">
                <button class="btn btn-xs btn-info" onclick="viewTicketDetail('${ticket.id}')">👁️ Detail</button>
                ${!ticket.assignedTo ? `
                  <button class="btn btn-xs btn-primary" onclick="assignTicketAction('${ticket.id}')">👨‍💼 Assign</button>
                ` : ''}
                ${ticket.status === 'open' || ticket.status === 'in_progress' ? `
                  <button class="btn btn-xs btn-success" onclick="resolveTicketAction('${ticket.id}')">✅ Resolve</button>
                ` : ''}
                ${ticket.status !== 'closed' ? `
                  <button class="btn btn-xs btn-outline-danger" onclick="closeTicketAction('${ticket.id}')">📋 Tutup</button>
                ` : ''}
              </div>
            </div>
          </div>
        </div>
      `;
    }).join('');
  }

  function filterHelpdeskTab(filter, btn) {
    document.querySelectorAll('.btn-toggle').forEach(b => b.classList.remove('active'));
    if (btn) btn.classList.add('active');

    let tickets;
    if (filter === 'sla') {
      tickets = TicketManager.getSLABreached();
    } else if (filter === 'all') {
      tickets = TicketManager.getTickets({});
    } else {
      tickets = TicketManager.getTickets({ status: filter });
    }

    document.getElementById('helpdesk-tickets-list').innerHTML = renderTicketsList(tickets.slice(0, 30));
  }

  /**
   * View ticket detail
   */
  function viewTicketDetail(ticketId) {
    const tickets = TicketManager.getTickets({});
    const ticket = tickets.find(t => t.id === ticketId);
    if (!ticket) return;

    const responsesHtml = ticket.responses.map(r => `
      <div class="card card-flat mb-2" style="${r.isStaff ? 'background: var(--color-info-bg);' : ''}">
        <div class="card-body">
          <div class="flex justify-between">
            <strong>${r.userName}</strong>
            <span class="text-xs text-muted">${formatRelativeDate(r.createdAt)}</span>
          </div>
          <p class="text-sm mt-1">${r.message}</p>
          ${r.attachment ? `<div class="text-xs mt-1"><a href="${r.attachment}" target="_blank">📎 Lampiran</a></div>` : ''}
        </div>
      </div>
    `).join('');

    showModalDetail(`🎫 ${ticket.ticketNumber} - ${ticket.subject}`, `
      <div class="mb-3 flex flex-wrap gap-2">
        <span class="badge badge-neutral">${ticket.category}</span>
        <span class="badge ${ticket.priority === 'urgent' ? 'badge-danger' : 'badge-warning'}">${ticket.priority}</span>
        <span class="badge ${ticket.status === 'open' ? 'badge-danger' : 'badge-info'}">${ticket.status.replace(/_/g, ' ')}</span>
      </div>
      <p class="text-sm"><strong>Pelapor:</strong> ${ticket.userName} (${ticket.userRole})</p>
      <p class="text-sm"><strong>Assigned:</strong> ${ticket.assignedName || 'Belum di-assign'}</p>
      <p class="text-sm"><strong>Dibuat:</strong> ${formatDateTime(ticket.createdAt)}</p>
      ${ticket.sla ? `
        <p class="text-sm"><strong>SLA Response:</strong> ${formatDateTime(ticket.sla.responseDeadline)} ${new Date(ticket.sla.responseDeadline) < new Date() ? '<span class="text-danger">(Breached!)</span>' : ''}</p>
      ` : ''}

      <hr class="my-3">

      <h4 class="font-semibold text-sm mb-2">💬 Percakapan (${ticket.responses.length})</h4>
      ${responsesHtml || '<p class="text-muted text-sm">Belum ada respon</p>'}

      <div class="mt-3">
        <textarea class="form-textarea" id="ticket-response-message" rows="2" placeholder="Tulis respon..."></textarea>
        <button class="btn btn-primary btn-sm mt-2" onclick="addTicketResponse('${ticketId}')">💬 Kirim Respon</button>
      </div>
    `);
  }

  function showCreateTicketForm() {
    const subject = prompt('Subjek tiket:');
    if (!subject) return;
    const message = prompt('Deskripsi masalah:');
    if (!message) return;
    const category = prompt(`Kategori (${HELPDESK_CONFIG.categories.join('/')}):`, 'Lainnya') || 'Lainnya';
    const priority = prompt(`Prioritas (${HELPDESK_CONFIG.priorities.join('/')}):`, 'medium') || 'medium';

    const user = getCurrentUser();
    TicketManager.createTicket({
      userId: user?.id || 'guest',
      userName: user?.nama || 'Guest',
      userRole: user?.role || 'guest',
      subject, message, category, priority
    });

    renderHelpdeskDashboard();
    showToast('Tiket dibuat!', 'success');
  }

  function assignTicketAction(ticketId) {

  const AssetManager = {
    /**
     * Tambah aset baru
     */
    addAsset(data) {
      const assets = getAssetData('assets');
      const {
        name, category = 'Lainnya', description = '', serialNumber = '',
        purchaseDate = new Date().toISOString(), purchasePrice = 0,
        usefulLife = ASSET_CONFIG.defaultUsefulLife, location = '', department = '',
        assignedTo = null, notes = '', supplier = ''
      } = data;

      const asset = {
        id: generateId('AST'),
        assetCode: `${ASSET_CONFIG.qrCodePrefix}${generateRandomString(8).toUpperCase()}`,
        name, category, description, serialNumber,
        purchaseDate, purchasePrice,
        usefulLife, // bulan
        monthlyDepreciation: usefulLife > 0 ? Math.round((purchasePrice / usefulLife) * 100) / 100 : 0,
        currentValue: purchasePrice,
        accumulatedDepreciation: 0,
        location, department,
        assignedTo,
        status: 'available',
        condition: 'good', // excellent, good, fair, poor
        notes, supplier,
        maintenanceRecords: [],
        borrowHistory: [],
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };

      assets.push(asset);
      saveAssetData('assets', assets);
      systemLogger.info('asset', `Asset created: ${asset.assetCode} - ${name}`);
      return asset;
    },

    /**
     * Update aset
     */
    updateAsset(assetId, data) {
      const assets = getAssetData('assets');
      const index = assets.findIndex(a => a.id === assetId);
      if (index === -1) return null;
      assets[index] = { ...assets[index], ...data, updatedAt: new Date().toISOString() };
      saveAssetData('assets', assets);
      return assets[index];
    },

    /**
     * Hitung penyusutan (depresiasi)
     */
    calculateDepreciation(assetId) {
      const asset = this.getAssets().find(a => a.id === assetId);
      if (!asset || asset.purchasePrice <= 0 || asset.usefulLife <= 0) return null;

      const monthsSincePurchase = Math.floor(
        (Date.now() - new Date(asset.purchaseDate).getTime()) / (30 * 86400000)
      );
      const monthsElapsed = Math.min(monthsSincePurchase, asset.usefulLife);
      const depreciation = Math.round(monthsElapsed * asset.monthlyDepreciation * 100) / 100;
      const currentValue = Math.max(0, Math.round((asset.purchasePrice - depreciation) * 100) / 100);

      this.updateAsset(assetId, {
        accumulatedDepreciation: depreciation,
        currentValue
      });

      return { monthsElapsed, depreciation, currentValue };
    },

    /**
     * Hitung penyusutan semua aset
     */
    calculateAllDepreciation() {
      const assets = this.getAssets();
      let totalDepreciation = 0;
      assets.forEach(a => {
        const result = this.calculateDepreciation(a.id);
        if (result) totalDepreciation += result.depreciation;
      });
      return { totalDepreciation, updatedAssets: assets.length };
    },

    /**
     * Tambah catatan maintenance
     */
    addMaintenance(assetId, data) {
      const assets = getAssetData('assets');
      const asset = assets.find(a => a.id === assetId);
      if (!asset) return null;

      const record = {
        id: generateId('MNT'),
        type: data.type || 'routine', // routine, repair, inspection
        description: data.description || '',
        cost: data.cost || 0,
        vendor: data.vendor || '',
        date: data.date || new Date().toISOString(),
        nextMaintenanceDate: data.nextMaintenanceDate || null,
        performedBy: data.performedBy || getCurrentUserId(),
        notes: data.notes || ''
      };

      asset.maintenanceRecords.push(record);
      asset.status = 'available'; // Kembalikan ke tersedia setelah maintenance
      asset.updatedAt = new Date().toISOString();
      saveAssetData('assets', assets);
      return record;
    },

    /**
     * Pinjamkan aset
     */
    borrowAsset(assetId, borrowerId, borrowDate = new Date().toISOString()) {
      const asset = this.getAssets().find(a => a.id === assetId);
      if (!asset) return { success: false, message: 'Aset tidak ditemukan' };
      if (asset.status !== 'available') return { success: false, message: 'Aset tidak tersedia' };

      const borrower = getUserById(borrowerId);
      const expectedReturn = new Date(new Date(borrowDate).getTime() + ASSET_CONFIG.maxBorrowDays * 86400000);

      const borrowRecord = {
        id: generateId('BRW'),
        borrowerId,
        borrowerName: borrower?.nama || 'Unknown',
        borrowDate,
        expectedReturnDate: expectedReturn.toISOString(),
        actualReturnDate: null,
        status: 'active', // active, returned, overdue
        conditionBefore: asset.condition,
        notes: ''
      };

      asset.borrowHistory.push(borrowRecord);
      asset.status = 'borrowed';
      asset.assignedTo = borrowerId;
      asset.updatedAt = new Date().toISOString();

      saveAssetData('assets', getAssetData('assets').map(a => a.id === assetId ? asset : a));
      return { success: true, borrowRecord, asset };
    },

    /**
     * Kembalikan aset
     */
    returnAsset(assetId, borrowId, conditionAfter = 'good', notes = '') {
      const assets = getAssetData('assets');
      const asset = assets.find(a => a.id === assetId);
      if (!asset) return null;

      const borrowRecord = asset.borrowHistory.find(b => b.id === borrowId);
      if (!borrowRecord) return null;

      borrowRecord.actualReturnDate = new Date().toISOString();
      borrowRecord.status = new Date(borrowRecord.actualReturnDate) > new Date(borrowRecord.expectedReturnDate) ? 'overdue' : 'returned';
      borrowRecord.conditionAfter = conditionAfter;
      borrowRecord.notes = notes;

      asset.status = 'available';
      asset.condition = conditionAfter;
      asset.assignedTo = null;
      asset.updatedAt = new Date().toISOString();
      saveAssetData('assets', assets);
      return borrowRecord;
    },

    /**
     * Dapatkan semua aset
     */
    getAssets(filters = {}) {
      let assets = getAssetData('assets');
      if (filters.category) assets = assets.filter(a => a.category === filters.category);
      if (filters.status) assets = assets.filter(a => a.status === filters.status);
      if (filters.department) assets = assets.filter(a => a.department === filters.department);
      if (filters.assignedTo) assets = assets.filter(a => a.assignedTo === filters.assignedTo);
      if (filters.search) {
        const s = filters.search.toLowerCase();
        assets = assets.filter(a => a.name.toLowerCase().includes(s) || a.assetCode.toLowerCase().includes(s) || a.serialNumber.toLowerCase().includes(s));
      }
      return assets.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    },

    /**
     * Dapatkan statistik aset
     */
    getStats() {
      const assets = this.getAssets();
      const stats = {
        total: assets.length,
        totalValue: assets.reduce((s, a) => s + a.purchasePrice, 0),
        currentValue: assets.reduce((s, a) => s + (a.currentValue || a.purchasePrice), 0),
        totalDepreciation: assets.reduce((s, a) => s + (a.accumulatedDepreciation || 0), 0),
        byStatus: {},
        byCategory: {},
        borrowed: assets.filter(a => a.status === 'borrowed').length,
        maintenanceDue: 0
      };

      ASSET_CONFIG.statuses.forEach(s => { stats.byStatus[s] = assets.filter(a => a.status === s).length; });
      ASSET_CONFIG.categories.forEach(c => { stats.byCategory[c] = assets.filter(a => a.category === c).length; });

      // Maintenance due dalam 30 hari
      const reminderDate = new Date(Date.now() + ASSET_CONFIG.maintenanceReminderDays * 86400000);
      assets.forEach(a => {
        a.maintenanceRecords.forEach(m => {
          if (m.nextMaintenanceDate && new Date(m.nextMaintenanceDate) <= reminderDate && new Date(m.nextMaintenanceDate) >= new Date()) {
            stats.maintenanceDue++;
          }
        });
      });

      return stats;
    }
  };

    /**

     */
    register(method, path, handler, options = {}) {
      const route = {
        id: generateId('API'),
        method: method.toUpperCase(),
        path: this.normalizePath(path),
        handler,
        options: {

      };

      // Check duplicate
      const exists = this.routes.find(r => r.method === route.method && r.path === route.path);
      if (exists) {
        systemLogger.warn('gateway', `Route already exists: ${method} ${path}, updating`);
        Object.assign(exists, route, { id: exists.id, createdAt: exists.createdAt });
        return exists;
      }

      this.routes.push(route);
      systemLogger.info('gateway', `Route registered: ${method} ${path}`);
      return route;
    }

    /**
     * Convenience methods
     */
    get(path, handler, options = {}) { return this.register('GET', path, handler, options); }
    post(path, handler, options = {}) { return this.register('POST', path, handler, options); }
    put(path, handler, options = {}) { return this.register('PUT', path, handler, options); }
    patch(path, handler, options = {}) { return this.register('PATCH', path, handler, options); }
    delete(path, handler, options = {}) { return this.register('DELETE', path, handler, options); }

    /**
     * Group routes dengan prefix
     */
    group(prefix, callback) {
      const previousPrefix = this._currentPrefix || '';
      this._currentPrefix = previousPrefix + this.normalizePath(prefix);
      callback(this);
      this._currentPrefix = previousPrefix;
    }

    /**
     * Add middleware
     */
    use(middleware) {
      this.middlewares.push(middleware);
    }

    /**
     * Normalize path
     */
    normalizePath(path) {
      let fullPath = (this._currentPrefix || '') + path;
      // Ensure starts with /
      if (!fullPath.startsWith('/')) fullPath = '/' + fullPath;
      // Remove trailing slash
      if (fullPath.length > 1 && fullPath.endsWith('/')) fullPath = fullPath.slice(0, -1);
      // Remove double slashes
      fullPath = fullPath.replace(/\/+/g, '/');
      return fullPath;
    }

    /**
     * Match route
     */
    matchRoute(method, path) {
      const normalizedPath = this.normalizePath(path);
      return this.routes.find(route => {
        if (route.method !== method.toUpperCase()) return false;
        return this.pathMatch(route.path, normalizedPath);
      });
    }

    /**
     * Path matching dengan parameter support
     * Contoh: /users/:id akan match /users/123
     */
    pathMatch(routePath, requestPath) {
      const routeParts = routePath.split('/');
      const requestParts = requestPath.split('/');

      if (routeParts.length !== requestParts.length) return false;

      const params = {};
      for (let i = 0; i < routeParts.length; i++) {
        if (routeParts[i].startsWith(':')) {
          // Parameter
          params[routeParts[i].substring(1)] = requestParts[i];
        } else if (routeParts[i] !== requestParts[i]) {
          return false;
        }
      }

      // Attach params untuk digunakan nanti
      this._lastMatchedParams = params;
      return true;
    }

     * Get last matched params
     */
    getMatchedParams() {
      return this._lastMatchedParams || {};
    }

     * Get all routes (for documentation)

    /**

    checkLimit(identifier, limit = GATEWAY_CONFIG.defaultRateLimit) {
      if (!GATEWAY_CONFIG.enableRateLimiting) return { allowed: true };

      const now = Date.now();
      if (!this.requests[identifier]) {
        this.requests[identifier] = [];
      }

      // Clean old entries (1 menit)
      this.requests[identifier] = this.requests[identifier].filter(t => now - t < 60000);

      if (this.requests[identifier].length >= limit) {
        const oldest = this.requests[identifier][0];
        const retryAfter = Math.ceil((60000 - (now - oldest)) / 1000);
        return {
          allowed: false,
          retryAfter,
          message: `Rate limit exceeded. Try again in ${retryAfter}s`
        };
      }

      this.requests[identifier].push(now);
      return { allowed: true };
    },

    /**
     * Get rate limit status
     */
    getStatus(identifier) {
      const now = Date.now();
      if (!this.requests[identifier]) {
        return { used: 0, limit: GATEWAY_CONFIG.defaultRateLimit, remaining: GATEWAY_CONFIG.defaultRateLimit };
      }
      this.requests[identifier] = this.requests[identifier].filter(t => now - t < 60000);
      const used = this.requests[identifier].length;

        used,
        limit: GATEWAY_CONFIG.defaultRateLimit,
        remaining: Math.max(0, GATEWAY_CONFIG.defaultRateLimit - used)

        }

        if (!route) {
          return ResponseFormatter.error(`Route not found: ${method} ${path}`, 404);

        // 4. Execute handler
        const params = apiRouter.getMatchedParams();
        const queryParams = this.parseQueryParams(options.query || '');

        const request = {
          method,
          path,
          params,
          query: queryParams,
          body,
          headers,
          apiKey,
          userId: authResult?.keyData?.name || 'api'
        };

        const result = await route.handler(request);

        // 5. Update stats
        const duration = performance.now() - startTime;
        route.stats.calls++;
        route.stats.avgResponseTime = Math.round(
          ((route.stats.avgResponseTime * (route.stats.calls - 1)) + duration) / route.stats.calls
        );
        route.stats.lastCalled = new Date().toISOString();

        GATEWAY_CONFIG.stats.totalRequests++;
        GATEWAY_CONFIG.stats.successfulRequests++;
        GATEWAY_CONFIG.stats.lastRequest = new Date().toISOString();

        return result;
      } catch (error) {
        const duration = performance.now() - startTime;
        GATEWAY_CONFIG.stats.totalRequests++;
        GATEWAY_CONFIG.stats.failedRequests++;

        const route = apiRouter.matchRoute(method, path);
        if (route) {
          route.stats.errors++;
        }

        systemLogger.error('gateway', `API Error: ${method} ${path}`, { error: error.message });
        return ResponseFormatter.error(error.message || 'Internal Server Error', 500);
      }

      const params = {};
      new URLSearchParams(queryString).forEach((value, key) => {
        // Try parse JSON

      });
      return params;

  const KBManager = {
    /**
     * Buat artikel baru
     */
    createArticle(data) {
      const articles = getKBData('articles');
      const {
        title, content = '', category = 'Umum', tags = [],
        authorId = getCurrentUserId(), authorName = getCurrentUser()?.nama || 'System',
        isPublished = false, isPublic = true, slug = ''
      } = data;

      // Generate slug

      // Cek duplicate slug
      if (articles.find(a => a.slug === articleSlug)) {
        return { success: false, message: 'Slug sudah digunakan, ganti judul' };
      }

      const article = {
        id: generateId('KB'),
        title, content, category, tags,
        slug: articleSlug,
        authorId, authorName,
        isPublished, isPublic,
        views: 0,
        helpfulCount: 0,

        revisions: [{
          id: generateId('REV'),

          editedAt: new Date().toISOString(),

        }],
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
        publishedAt: isPublished ? new Date().toISOString() : null
      };

      articles.push(article);
      saveKBData('articles', articles);
      systemLogger.info('kb', `Article created: ${title}`);
      return { success: true, article };
    },

    /**
     * Update artikel (dengan revision tracking)
     */
    updateArticle(articleId, data, changeNote = 'Artikel diperbarui') {
      const articles = getKBData('articles');
      const index = articles.findIndex(a => a.id === articleId);
      if (index === -1) return null;

      const editorId = getCurrentUserId();
      const editorName = getCurrentUser()?.nama || 'System';

      // Simpan revisi jika konten berubah
      if (data.content && data.content !== articles[index].content) {
        const revision = {
          id: generateId('REV'),
          content: articles[index].content,
          editedBy: editorId,
          editedByName: editorName,
          editedAt: new Date().toISOString(),
          changeNote: changeNote
        };
        articles[index].revisions.push(revision);

        // Batasi jumlah revisi
        if (articles[index].revisions.length > KB_CONFIG.maxRevisionsPerArticle) {
          articles[index].revisions.splice(0, articles[index].revisions.length - KB_CONFIG.maxRevisionsPerArticle);
        }
      }

      // Update data
      articles[index] = {
        ...articles[index],
        ...data,
        updatedAt: new Date().toISOString()
      };

      // Auto-set publishedAt jika baru dipublish
      if (data.isPublished && !articles[index].publishedAt) {
        articles[index].publishedAt = new Date().toISOString();
      }

      saveKBData('articles', articles);
      return articles[index];
    },

    /**
     * Hapus artikel
     */
    deleteArticle(articleId) {
      const articles = getKBData('articles');
      const filtered = articles.filter(a => a.id !== articleId);
      if (filtered.length === articles.length) return false;
      saveKBData('articles', filtered);
      return true;
    },

      const articles = getKBData('articles');

      if (!article) return;

      saveKBData('articles', articles);

    recordFeedback(articleId, isHelpful) {
      const articles = getKBData('articles');
      const article = articles.find(a => a.id === articleId);
      if (!article) return;

      if (isHelpful) {
        article.helpfulCount = (article.helpfulCount || 0) + 1;
      } else {
        article.unhelpfulCount = (article.unhelpfulCount || 0) + 1;
      }
      saveKBData('articles', articles);
    },

    /**
     * Dapatkan semua artikel
     */
    getArticles(filters = {}) {
      let articles = getKBData('articles');

      if (filters.category && filters.category !== 'all') {
        articles = articles.filter(a => a.category === filters.category);
      }
      if (filters.isPublished !== undefined) {
        articles = articles.filter(a => a.isPublished === filters.isPublished);
      }
      if (filters.isPublic !== undefined) {
        articles = articles.filter(a => a.isPublic === filters.isPublic);
      }
      if (filters.authorId) {
        articles = articles.filter(a => a.authorId === filters.authorId);
      }
      if (filters.tag) {
        articles = articles.filter(a => (a.tags || []).includes(filters.tag));
      }
      if (filters.search) {
        const s = filters.search.toLowerCase();
        articles = articles.filter(a =>
          a.title.toLowerCase().includes(s) ||
          a.content.toLowerCase().includes(s) ||
          (a.tags || []).some(tag => tag.toLowerCase().includes(s))
        );
      }

      return articles.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    },

    /**
     * Dapatkan artikel berdasarkan ID
     */
    getArticleById(articleId) {
      return getKBData('articles').find(a => a.id === articleId) || null;
    },

    /**
     * Dapatkan artikel berdasarkan slug
     */
    getArticleBySlug(slug) {
      return getKBData('articles').find(a => a.slug === slug) || null;
    },

    /**
     * Dapatkan artikel terkait (berdasarkan tag & kategori)
     */
    getRelatedArticles(articleId, limit = KB_CONFIG.relatedArticlesCount) {
      const article = this.getArticleById(articleId);
      if (!article) return [];

      const allArticles = this.getArticles({ isPublished: true });
      const scored = allArticles
        .filter(a => a.id !== articleId)
        .map(a => {
          let score = 0;
          // Kategori sama
          if (a.category === article.category) score += 3;
          // Tag yang sama
          const commonTags = (a.tags || []).filter(tag => (article.tags || []).includes(tag));
          score += commonTags.length * 2;
          // Popularitas
          score += Math.min(a.views / 100, 2);
          return { ...a, relatedScore: score };
        });

      return scored
        .sort((a, b) => b.relatedScore - a.relatedScore)
        .slice(0, limit);
    },

    /**
     * Dapatkan artikel populer
     */
    getPopularArticles(limit = KB_CONFIG.popularArticlesCount) {
      return this.getArticles({ isPublished: true })
        .sort((a, b) => b.views - a.views)
        .slice(0, limit);
    },

        popularArticles: published.sort((a, b) => b.views - a.views).slice(0, 5)
      };
    }

      .replace(/`(.+?)`/g, '<code style="background: var(--bg-secondary); padding: 1px 4px; border-radius: 3px;">$1</code>')

      // Hindari duplikat

      if (existing) {
        existing.operation = operation;
        existing.timestamp = Date.now();
      } else {
        this.syncQueue.push({ key, operation, timestamp: Date.now() });
      }

      // Batasi queue
      if (this.syncQueue.length > 100) {
        this.syncQueue.splice(0, this.syncQueue.length - 100);
      }

      // Proses queue jika tidak sedang sync
      if (!this.isSyncing) {
        this.processSyncQueue();
      }

    /**
     * Proses sync queue
     */
    async processSyncQueue() {
      if (this.isSyncing || this.syncQueue.length === 0) return;
      this.isSyncing = true;

      const batch = this.syncQueue.splice(0, 10); // 10 items per batch
      const syncData = {};

      batch.forEach(item => {
        const value = localStorage.getItem(item.key);
        if (value) {
          try {
            syncData[item.key] = JSON.parse(value);
          } catch {
            syncData[item.key] = value;
          }
        } else {
          syncData[item.key] = null; // Tandai untuk dihapus
        }
      });

      // Simpan ke cloud
      await this.uploadToCloud(syncData, 'incremental');

      this.isSyncing = false;

      // Lanjutkan jika masih ada
      if (this.syncQueue.length > 0) {
        setTimeout(() => this.processSyncQueue(), 1000);
      }

    async restoreFromCloud(backupId, provider = 'local') {
      let data;

      if (provider === 'local') {
        const key = APP_CONFIG.storagePrefix + 'cloudsync_backup_full';
        data = getFromStorage(key, null);
      } else {
        // Simulasi download dari cloud
        data = await this.downloadFromCloud(backupId, provider);
      }

      if (!data) {
        return { success: false, message: 'Backup tidak ditemukan' };
      }

      // Dekripsi jika perlu
      if (CLOUD_SYNC_CONFIG.encryptionEnabled && CLOUD_SYNC_CONFIG.encryptionKey) {
        data = await this.decryptData(data);
      }

      // Dekompresi jika perlu
      if (CLOUD_SYNC_CONFIG.compressionEnabled && data._compressed) {
        data = await this.decompressData(data);
      }

      // Restore data
      const keys = data.data ? Object.keys(data.data) : Object.keys(data);
      let restored = 0;

      if (data.data) {
        // Full backup format
        Object.entries(data.data).forEach(([key, value]) => {
          const serialized = typeof value === 'string' ? value : JSON.stringify(value);
          localStorage.setItem(key, serialized);
          restored++;
        });
      } else {
        // Raw format
        Object.entries(data).forEach(([key, value]) => {
          if (key !== 'id' && key !== 'timestamp' && key !== 'version' && key !== '_compressed') {
            const serialized = typeof value === 'string' ? value : JSON.stringify(value);
            localStorage.setItem(key, serialized);
            restored++;
          }
        });
      }

      systemLogger.info('cloudsync', `Restored ${restored} keys from backup ${backupId}`);
      return { success: true, restored };
    },

function restoreAllDataJSON(input) {
  var file = input.files[0];
  if (!file) return;
  if (!confirm('⚠️ SEMUA data saat ini akan ditimpa. Lanjutkan?')) return;
  var reader = new FileReader();
  reader.onload = function(e) {
    try {
      var data = JSON.parse(e.target.result);
      localStorage.clear();
      Object.keys(data).forEach(function(key) {
        localStorage.setItem(key, data[key]);
      });
      showToast('✅ Data JSON berhasil di-restore! Refresh...', 'success');
      setTimeout(function() { location.reload(); }, 2000);
    } catch (err) {
      showToast('❌ File JSON tidak valid.', 'error');
    }
  };
  reader.readAsText(file);
}

function restoreFromHTML(input) {
  var file = input.files[0];
  if (!file) return;
  if (!confirm('⚠️ Halaman akan diganti dengan file HTML ini. Lanjutkan?')) return;
  var reader = new FileReader();
  reader.onload = function(e) {
    var match = e.target.result.match(/<html[^>]*>([\s\S]*)<\/html>/i);
    if (match) {
      document.documentElement.innerHTML = match[1];
      showToast('✅ Halaman dipulihkan! Refresh...', 'success');
      setTimeout(function() { location.reload(); }, 1500);
    } else {
      document.open();
      document.write(e.target.result);
      document.close();
      showToast('✅ Halaman dipulihkan! Refresh...', 'success');
      setTimeout(function() { location.reload(); }, 1500);
    }
  };
  reader.readAsText(file);
}

function loadAdminChat(username) {
  var allMessages = getChatMessages();
  var filtered = [];
  for (var i = 0; i < allMessages.length; i++) {
    var m = allMessages[i];
    if ((m.from === 'admin' && m.to === username) || (m.from === username && m.to === 'admin')) {
      filtered.push(m);
    }
  }
  var container = document.getElementById('adminChatMessages');
  if (!container) return;
  if (filtered.length === 0) {
    container.innerHTML = '<p style="color:var(--text-secondary);text-align:center;padding:20px;">Belum ada chat dengan '+username+'.</p>';
    return;
  }
  filtered.sort(function(a, b) { return new Date(a.time) - new Date(b.time); });
  var html = '';
  for (var i = 0; i < filtered.length; i++) {
    var m = filtered[i];
    var isAdmin = (m.from === 'admin');
    var cls = isAdmin ? 'chat-msg admin' : 'chat-msg user';
    var sender = isAdmin ? 'Admin' : m.from;
    html += '<div class="'+cls+'"><div class="chat-sender">'+sender+'</div>'+m.text+'<div class="chat-time">'+new Date(m.time).toLocaleString()+'</div></div>';
  }
  container.innerHTML = html;
  container.scrollTop = container.scrollHeight;
}

function sendAdminChat() {
  var select = document.getElementById('adminChatUserSelect');
  var target = select ? select.value : '';
  if (!target) { showToast('Pilih user!','error'); return; }
  var input = document.getElementById('adminChatInput');
  var text = input.value.trim();
  if (!text) return;
  var msgs = getChatMessages();
  msgs.push({ from: 'admin', to: target, text: text, time: new Date().toISOString() });
  saveChatMessages(msgs);
  input.value = '';
  loadAdminChat(target);
  showToast('Balasan terkirim ke '+target, 'success');
}

  var text = textarea.value.trim();
  if (!text) { showToast('Tulis pesan!','error'); return; }
  var users = getAllUsers();

  for (var i = 0; i < users.length; i++) {

  saveChatMessages(msgs);
  textarea.value = '';
  showToast('Pesan massal terkirim ke '+count+' user!', 'success');
}

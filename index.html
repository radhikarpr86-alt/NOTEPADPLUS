function renderGuestChat() {
  var messages = getChatMessages().filter(function(message) { return message.to === 'admin' || message.from === 'guest'; });
  var chatBox = document.getElementById('guestChatMessages');
  if (!chatBox) return;

      '<div class="chat-sender">' + (isGuest ? 'Anda' : 'Admin') + '</div>' +
      message.text +
      '<div class="chat-time">' + new Date(message.time).toLocaleString() + '</div>' +
    '</div>';
  }).join('');

}

  if (deferredPrompt) {
    deferredPrompt.prompt();
    deferredPrompt.userChoice.then(function(choiceResult) {
      if (choiceResult.outcome === 'accepted') showToast('Instalasi berhasil!', 'success');
      deferredPrompt = null;
    });
  } else if (window.matchMedia('(display-mode: standalone)').matches) {
    showToast('Aplikasi sudah terinstall.', 'warning');
  } else {
    alert('Buka menu browser lalu pilih "Tambahkan ke Layar Utama" atau "Install".');
  }
}

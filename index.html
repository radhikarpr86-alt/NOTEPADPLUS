<script id="vendor-procurement-system">
  // ==========================================
  // KONFIGURASI PROCUREMENT
  // ==========================================
  const PROCUREMENT_CONFIG = {
    storagePrefix: APP_CONFIG.storagePrefix + 'proc_',
    maxVendors: 200,
    maxPurchaseReqs: 500,
    maxPurchaseOrders: 500,
    vendorStatuses: ['active', 'inactive', 'blacklisted', 'under_review'],
    prStatuses: ['draft', 'submitted', 'approved', 'rejected', 'converted'],
    poStatuses: ['draft', 'sent', 'confirmed', 'partially_received', 'received', 'invoiced', 'paid', 'cancelled'],
    paymentTerms: ['Tunai', '7 Hari', '14 Hari', '30 Hari', '60 Hari', '90 Hari'],
    approvalLevels: ['manager', 'director', 'finance'],
    evaluationCriteria: ['quality', 'price', 'delivery', 'service', 'compliance'],
    maxEvaluationScore: 5
  };

  // ==========================================
  // DATA STORE
  // ==========================================

  function getProcData(key) {
    return getFromStorage(PROCUREMENT_CONFIG.storagePrefix + key, []);
  }

  function saveProcData(key, data) {
    return saveToStorage(PROCUREMENT_CONFIG.storagePrefix + key, data);
  }

  // ==========================================
  // VENDOR MANAGER
  // ==========================================

  const ProcurementVendorManager = {
    /**
     * Registrasi vendor baru
     */
    registerVendor(data) {
      const vendors = getProcData('vendors');
      const {
        companyName, contactPerson = '', email = '', phone = '', address = '',
        taxId = '', categories = [], bankName = '', bankAccount = '',
        paymentTerms = '30 Hari', notes = ''
      } = data;

      if (!companyName) return { success: false, message: 'Nama perusahaan wajib diisi' };
      if (vendors.length >= PROCUREMENT_CONFIG.maxVendors) {
        return { success: false, message: `Maksimal ${PROCUREMENT_CONFIG.maxVendors} vendor` };
      }

      const vendor = {
        id: generateId('VND'),
        vendorCode: `VND-${String(vendors.length + 1).padStart(4, '0')}`,
        companyName, contactPerson, email, phone, address, taxId,
        categories, bankName, bankAccount, paymentTerms, notes,
        status: 'active',
        totalOrders: 0,
        totalSpent: 0,
        evaluations: [],
        averageScore: 0,
        lastOrderAt: null,
        createdAt: new Date().toISOString()
      };

      vendors.push(vendor);
      saveProcData('vendors', vendors);
      return { success: true, vendor };
    },

    /**
     * Update vendor
     */
    updateVendor(vendorId, data) {
      const vendors = getProcData('vendors');
      const index = vendors.findIndex(v => v.id === vendorId);
      if (index === -1) return null;
      vendors[index] = { ...vendors[index], ...data, updatedAt: new Date().toISOString() };
      saveProcData('vendors', vendors);
      return vendors[index];
    },

    /**
     * Evaluasi vendor
     */
    evaluateVendor(vendorId, scores = {}, notes = '') {
      const vendors = getProcData('vendors');
      const vendor = vendors.find(v => v.id === vendorId);
      if (!vendor) return null;

      const evaluation = {
        id: generateId('EVL'),
        scores,
        notes,
        evaluatedBy: getCurrentUserId(),
        evaluatedAt: new Date().toISOString(),
        averageScore: Object.values(scores).length > 0
          ? Math.round((Object.values(scores).reduce((a, b) => a + b, 0) / Object.values(scores).length) * 10) / 10
          : 0
      };

      vendor.evaluations.push(evaluation);
      const totalAvg = vendor.evaluations.reduce((s, e) => s + e.averageScore, 0) / vendor.evaluations.length;
      vendor.averageScore = Math.round(totalAvg * 10) / 10;
      saveProcData('vendors', vendors);
      return evaluation;
    },

    getVendors(filters = {}) {
      let vendors = getProcData('vendors');
      if (filters.status) vendors = vendors.filter(v => v.status === filters.status);
      if (filters.category) vendors = vendors.filter(v => (v.categories || []).includes(filters.category));
      if (filters.search) {
        const s = filters.search.toLowerCase();
        vendors = vendors.filter(v => v.companyName.toLowerCase().includes(s) || v.vendorCode.toLowerCase().includes(s));
      }
      return vendors.sort((a, b) => b.averageScore - a.averageScore);
    },

    getVendorById(vendorId) {
      return getProcData('vendors').find(v => v.id === vendorId) || null;
    }
  };

  // ==========================================
  // PURCHASE REQUISITION MANAGER
  // ==========================================

  const PRManager = {
    /**
     * Buat Purchase Requisition
     */
    createPR(data) {
      const prs = getProcData('purchase_requisitions');
      const {
        title, description = '', department = '', requesterId = getCurrentUserId(),
        requesterName = getCurrentUser()?.nama || 'Requester',
        items = [], priority = 'normal', requiredDate = null, notes = ''
      } = data;

      if (items.length === 0) return { success: false, message: 'Minimal satu item diperlukan' };
      if (prs.length >= PROCUREMENT_CONFIG.maxPurchaseReqs) {
        return { success: false, message: `Maksimal ${PROCUREMENT_CONFIG.maxPurchaseReqs} PR` };
      }

      const totalAmount = items.reduce((s, i) => s + (i.estimatedPrice || 0) * (i.quantity || 1), 0);

      const pr = {
        id: generateId('PRQ'),
        prNumber: `PR-${formatDate(new Date(), 'YYYYMMDD')}-${String(prs.length + 1).padStart(3, '0')}`,
        title, description, department, requesterId, requesterName,
        items: items.map((item, i) => ({
          id: `pri_${i + 1}`,
          name: item.name,
          description: item.description || '',
          quantity: item.quantity || 1,
          unit: item.unit || 'Pcs',
          estimatedPrice: item.estimatedPrice || 0,
          specifications: item.specifications || ''
        })),
        priority, requiredDate, notes,
        totalAmount,
        status: 'draft',
        approvedBy: null, approvedAt: null, rejectionReason: '',
        createdAt: new Date().toISOString(), updatedAt: new Date().toISOString()
      };

      prs.push(pr);
      saveProcData('purchase_requisitions', prs);
      return { success: true, pr };
    },

    /**
     * Submit PR untuk approval
     */
    submitPR(prId) {
      const prs = getProcData('purchase_requisitions');
      const pr = prs.find(p => p.id === prId);
      if (!pr || pr.status !== 'draft') return null;
      pr.status = 'submitted';
      pr.submittedAt = new Date().toISOString();
      pr.updatedAt = new Date().toISOString();
      saveProcData('purchase_requisitions', prs);
      return pr;
    },

    /**
     * Approve/reject PR
     */
    processPR(prId, isApproved, approverId = getCurrentUserId(), reason = '') {
      const prs = getProcData('purchase_requisitions');
      const pr = prs.find(p => p.id === prId);
      if (!pr || pr.status !== 'submitted') return null;

      pr.status = isApproved ? 'approved' : 'rejected';
      pr.approvedBy = approverId;
      pr.approvedAt = new Date().toISOString();
      if (!isApproved) pr.rejectionReason = reason;
      pr.updatedAt = new Date().toISOString();
      saveProcData('purchase_requisitions', prs);
      return pr;
    },

    getPRs(filters = {}) {
      let prs = getProcData('purchase_requisitions');
      if (filters.status && filters.status !== 'all') prs = prs.filter(p => p.status === filters.status);
      return prs.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    }
  };

  // ==========================================
  // PURCHASE ORDER MANAGER
  // ==========================================

  const POManager = {
    /**
     * Buat Purchase Order dari PR yang disetujui
     */
    createPOFromPR(prId, vendorId) {
      const prs = getProcData('purchase_requisitions');
      const pr = prs.find(p => p.id === prId);
      if (!pr || pr.status !== 'approved') return { success: false, message: 'PR belum disetujui' };

      const vendor = ProcurementVendorManager.getVendorById(vendorId);
      if (!vendor) return { success: false, message: 'Vendor tidak ditemukan' };

      const pos = getProcData('purchase_orders');
      if (pos.length >= PROCUREMENT_CONFIG.maxPurchaseOrders) {
        return { success: false, message: `Maksimal ${PROCUREMENT_CONFIG.maxPurchaseOrders} PO` };
      }

      const po = {
        id: generateId('PUR'),
        poNumber: `PO-${formatDate(new Date(), 'YYYYMMDD')}-${String(pos.length + 1).padStart(3, '0')}`,
        prId, prNumber: pr.prNumber,
        vendorId, vendorName: vendor.companyName,
        items: pr.items.map(item => ({
          ...item,
          receivedQty: 0,
          unitPrice: item.estimatedPrice || 0,
          totalPrice: (item.estimatedPrice || 0) * (item.quantity || 1)
        })),
        totalAmount: pr.totalAmount,
        paymentTerms: vendor.paymentTerms || '30 Hari',
        deliveryDate: pr.requiredDate,
        status: 'draft',
        receivedAt: null,
        invoicedAt: null,
        paidAt: null,
        notes: '',
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };

      // Update PR status
      pr.status = 'converted';
      pr.updatedAt = new Date().toISOString();
      saveProcData('purchase_requisitions', prs);

      pos.push(po);
      saveProcData('purchase_orders', pos);

      // Update vendor stats
      vendor.totalOrders = (vendor.totalOrders || 0) + 1;
      vendor.lastOrderAt = new Date().toISOString();
      saveProcData('vendors', getProcData('vendors'));

      return { success: true, po };
    },

    /**
     * Kirim PO ke vendor
     */
    sendPO(poId) {
      const pos = getProcData('purchase_orders');
      const po = pos.find(p => p.id === poId);
      if (!po || po.status !== 'draft') return null;
      po.status = 'sent';
      po.sentAt = new Date().toISOString();
      po.updatedAt = new Date().toISOString();
      saveProcData('purchase_orders', pos);
      return po;
    },

    /**
     * Terima barang (Goods Receipt)
     */
    receiveGoods(poId, receivedItems = []) {
      const pos = getProcData('purchase_orders');
      const po = pos.find(p => p.id === poId);
      if (!po || !['sent', 'confirmed', 'partially_received'].includes(po.status)) return null;

      receivedItems.forEach(ri => {
        const item = po.items.find(i => i.id === ri.itemId);
        if (item) {
          item.receivedQty = (item.receivedQty || 0) + (ri.quantity || 0);
        }
      });

      const allReceived = po.items.every(i => i.receivedQty >= i.quantity);
      po.status = allReceived ? 'received' : 'partially_received';
      if (allReceived) po.receivedAt = new Date().toISOString();

      po.updatedAt = new Date().toISOString();
      saveProcData('purchase_orders', pos);

      // Update stok produk jika ada produk terkait
      if (allReceived) {
        po.items.forEach(item => {
          if (item.productId) {
            updateStokProduk(item.productId, item.quantity);
            BatchManager.addBatch(item.productId, item.quantity, item.unitPrice || 0, {
              purchaseOrderId: poId,
              reason: 'Purchase Order',
              reference: po.poNumber
            });
          }
        });
      }

      return po;
    },

    /**
     * Update status PO
     */
    updatePOStatus(poId, status, notes = '') {
      const pos = getProcData('purchase_orders');
      const po = pos.find(p => p.id === poId);
      if (!po) return null;

      po.status = status;
      po.updatedAt = new Date().toISOString();
      if (status === 'paid') po.paidAt = new Date().toISOString();
      if (status === 'invoiced') po.invoicedAt = new Date().toISOString();
      if (notes) po.notes = notes;

      // Update vendor spending
      if (status === 'paid') {
        const vendors = getProcData('vendors');
        const vendor = vendors.find(v => v.id === po.vendorId);
        if (vendor) {
          vendor.totalSpent = (vendor.totalSpent || 0) + po.totalAmount;
          saveProcData('vendors', vendors);
        }
      }

      saveProcData('purchase_orders', pos);
      return po;
    },

    getPOs(filters = {}) {
      let pos = getProcData('purchase_orders');
      if (filters.status && filters.status !== 'all') pos = pos.filter(p => p.status === filters.status);
      if (filters.vendorId) pos = pos.filter(p => p.vendorId === filters.vendorId);
      return pos.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    },

    getPOById(poId) {
      return getProcData('purchase_orders').find(p => p.id === poId) || null;
    }
  };

  function showCreateVendorForm() {
    const companyName = prompt('Nama perusahaan:');
    if (!companyName) return;
    const contactPerson = prompt('Kontak person:') || '';
    const email = prompt('Email:') || '';
    const phone = prompt('Telepon:') || '';
    ProcurementVendorManager.registerVendor({ companyName, contactPerson, email, phone });
    renderProcurementDashboard();
    showToast('Vendor ditambahkan!', 'success');
  }

  function showCreatePRForm() {
    const title = prompt('Judul PR:');
    if (!title) return;
    const itemCount = parseInt(prompt('Jumlah item:', '2')) || 2;
    const items = [];
    for (let i = 0; i < itemCount; i++) {
      const name = prompt(`Item ${i + 1} - Nama:`) || `Item ${i + 1}`;
      const qty = parseInt(prompt(`Item ${i + 1} - Qty:`, '1')) || 1;
      const price = parseInt(prompt(`Item ${i + 1} - Estimasi harga:`, '0')) || 0;
      items.push({ name, quantity: qty, estimatedPrice: price });
    }
    PRManager.createPR({ title, items });
    renderProcurementDashboard();
    showToast('PR dibuat!', 'success');
  }

  function viewPRDetail(prId) {

    if (!pr) return;
    const vendors = ProcurementVendorManager.getVendors({ status: 'active' });

    const itemsHtml = pr.items.map(i => `<tr><td>${i.name}</td><td>${i.quantity} ${i.unit}</td><td>${formatCurrency(i.estimatedPrice)}</td><td>${formatCurrency(i.estimatedPrice * i.quantity)}</td></tr>`).join('');

    showModalDetail(`📋 ${pr.prNumber}: ${pr.title}`, `
      <div class="flex flex-wrap gap-2 mb-3"><span class="badge ${pr.status === 'approved' ? 'badge-success' : 'badge-warning'}">${pr.status}</span><span class="text-sm">Total: ${formatCurrency(pr.totalAmount)}</span></div>
      <table class="table table-compact mb-3"><thead><tr><th>Item</th><th>Qty</th><th>Harga</th><th>Subtotal</th></tr></thead><tbody>${itemsHtml}</tbody></table>
      ${pr.status === 'submitted' ? `<div class="flex gap-2 mb-3"><button class="btn btn-sm btn-success" onclick="processPRAction('${prId}', true)">✅ Approve</button><button class="btn btn-sm btn-danger" onclick="processPRAction('${prId}', false)">❌ Reject</button></div>` : ''}
      ${pr.status === 'approved' ? `
        <div class="form-group"><label class="form-label">Pilih Vendor</label>
          <select class="form-select" id="po-vendor-select">${vendors.map(v => `<option value="${v.id}">${v.companyName} (⭐${v.averageScore})</option>`).join('')}</select>
          <button class="btn btn-sm btn-primary mt-2" onclick="createPOAction('${prId}')">📦 Buat PO</button>
        </div>
      ` : ''}
    `);
  }

  function processPRAction(prId, isApproved) {
    const reason = !isApproved ? prompt('Alasan penolakan:') : '';
    PRManager.processPR(prId, isApproved, getCurrentUserId(), reason || '');
    closeModal('modal-detail');
    renderProcurementDashboard();
    showToast(isApproved ? 'PR disetujui!' : 'PR ditolak!', isApproved ? 'success' : 'warning');
  }

  function createPOAction(prId) {
    const vendorId = document.getElementById('po-vendor-select')?.value;
    if (!vendorId) return;
    const result = POManager.createPOFromPR(prId, vendorId);
    if (result.success) {
      closeModal('modal-detail');
      renderProcurementDashboard();
      showToast('PO berhasil dibuat!', 'success');
    } else {
      showToast(result.message, 'error');
    }
  }

  function viewPODetail(poId) {
    const po = POManager.getPOById(poId);
    if (!po) return;

    const itemsHtml = po.items.map(i => `<tr><td>${i.name}</td><td>${i.receivedQty}/${i.quantity} ${i.unit}</td><td>${formatCurrency(i.unitPrice)}</td><td>${formatCurrency(i.totalPrice)}</td></tr>`).join('');

    showModalDetail(`📦 ${po.poNumber}`, `
      <div class="flex flex-wrap gap-2 mb-3"><span class="badge ${po.status === 'received' ? 'badge-success' : 'badge-info'}">${po.status}</span><span class="text-sm">Vendor: ${po.vendorName}</span><span class="text-sm ml-2">Total: ${formatCurrency(po.totalAmount)}</span></div>
      <table class="table table-compact mb-3"><thead><tr><th>Item</th><th>Qty</th><th>Harga</th><th>Subtotal</th></tr></thead><tbody>${itemsHtml}</tbody></table>
      <div class="flex gap-2">

    maxTaxTypes: 20,
    taxTypes: [
      { id: 'ppn', name: 'PPN (Pajak Pertambahan Nilai)', rate: 11, category: 'sales', description: 'Pajak atas konsumsi barang dan jasa' },

  // ==========================================
  // TAX CALCULATOR

  const TaxCalculator = {
    /**
     * Hitung PPN dari jumlah
     */
    calculatePPN(amount, includeTax = false) {
      const ppnRate = (TAX_CONFIG.taxTypes.find(t => t.id === 'ppn')?.rate || 11) / 100;

      if (includeTax) {
        // Harga sudah termasuk PPN
        const baseAmount = amount / (1 + ppnRate);
        return {
          baseAmount: Math.round(baseAmount),
          taxAmount: Math.round(amount - baseAmount),
          totalAmount: amount,
          rate: ppnRate * 100
        };
      } else {
        // Harga belum termasuk PPN
        const taxAmount = Math.round(amount * ppnRate);
        return {
          baseAmount: amount,
          taxAmount,
          totalAmount: amount + taxAmount,
          rate: ppnRate * 100
        };
      }
    },

    /**
     * Hitung PPh 21 progresif (simulasi sederhana)
     */
    calculatePPh21(annualIncome) {
      // Tarif progresif PPh 21 (simulasi)
      const brackets = [
        { max: 60000000, rate: 0.05 },
        { max: 250000000, rate: 0.15 },
        { max: 500000000, rate: 0.25 },
        { max: 5000000000, rate: 0.30 },
        { max: Infinity, rate: 0.35 }
      ];

      let remaining = annualIncome;
      let totalTax = 0;
      let previousMax = 0;
      const breakdown = [];

      for (const bracket of brackets) {
        const taxableInBracket = Math.min(remaining, bracket.max - previousMax);
        if (taxableInBracket <= 0) break;

        const taxInBracket = Math.round(taxableInBracket * bracket.rate);
        totalTax += taxInBracket;
        breakdown.push({
          bracket: `${formatCurrency(previousMax)} - ${bracket.max === Infinity ? 'Tak Terbatas' : formatCurrency(bracket.max)}`,
          rate: bracket.rate * 100,
          taxable: taxableInBracket,
          tax: taxInBracket
        });

        remaining -= taxableInBracket;
        previousMax = bracket.max;
      }

      return {
        annualIncome,
        totalTax,
        monthlyTax: Math.round(totalTax / 12),
        effectiveRate: Math.round((totalTax / annualIncome) * 10000) / 100,
        breakdown
      };
    },

    /**
     * Hitung PPh Final UMKM
     */
    calculatePPhFinal(grossIncome) {
      const rate = (TAX_CONFIG.taxTypes.find(t => t.id === 'pph_final')?.rate || 0.5) / 100;
      const taxAmount = Math.round(grossIncome * rate);

      return {
        grossIncome,
        rate: rate * 100,
        taxAmount,
        netIncome: grossIncome - taxAmount
      };
    },

    /**
     * Hitung denda keterlambatan
     */
    calculatePenalty(taxAmount, dueDate, paymentDate = new Date()) {
      const due = new Date(dueDate);
      const paid = new Date(paymentDate);
      if (paid <= due) return { penalty: 0, monthsLate: 0 };

      const monthsLate = Math.min(
        Math.ceil((paid - due) / (30 * 86400000)),
        TAX_CONFIG.maxPenaltyMonths
      );
      const penaltyRate = TAX_CONFIG.penaltyRate / 100;
      const penalty = Math.round(taxAmount * penaltyRate * monthsLate);

      return {
        penalty,
        monthsLate,
        penaltyRate: TAX_CONFIG.penaltyRate,
        totalDue: taxAmount + penalty
      };
    }
  };

  const TaxManager = {
    /**
     * Buat catatan pajak baru
     */
    createTaxRecord(data) {
      const records = getTaxData('records');
      const {
        taxType = 'ppn', period = '', description = '', baseAmount = 0,
        dueDate = null, status = 'pending', notes = '', documents = []
      } = data;

      if (records.length >= TAX_CONFIG.maxTaxRecords) {
        return { success: false, message: `Maksimal ${TAX_CONFIG.maxTaxRecords} catatan pajak` };
      }

      const taxConfig = TAX_CONFIG.taxTypes.find(t => t.id === taxType);
      let calculation;

      // Hitung pajak berdasarkan tipe
      switch (taxType) {
        case 'ppn':
          calculation = TaxCalculator.calculatePPN(baseAmount);
          break;
        case 'pph_final':
          calculation = TaxCalculator.calculatePPhFinal(baseAmount);
          break;
        case 'pph21':
          calculation = TaxCalculator.calculatePPh21(baseAmount);
          break;
        default:
          calculation = {
            baseAmount,
            taxAmount: Math.round(baseAmount * ((taxConfig?.rate || 0) / 100)),
            rate: taxConfig?.rate || 0
          };
      }

      const record = {
        id: generateId('TAX'),
        taxCode: `TAX-${formatDate(new Date(), 'YYYYMMDD')}-${String(records.length + 1).padStart(3, '0')}`,
        taxType, period, description, baseAmount,
        taxAmount: calculation.taxAmount,
        totalAmount: calculation.totalAmount || (baseAmount + calculation.taxAmount),
        dueDate: dueDate || new Date(Date.now() + 30 * 86400000).toISOString(),
        status, notes, documents,
        payments: [],
        totalPaid: 0,
        penalty: 0,
        calculation,
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };

      records.push(record);
      saveTaxData('records', records);
      return { success: true, record };

     * Catat pembayaran pajak

    recordPayment(taxId, amount, date = new Date().toISOString(), method = 'transfer', reference = '') {
      const records = getTaxData('records');
      const record = records.find(r => r.id === taxId);
      if (!record) return null;

      const payment = {
        id: generateId('TXP'),
        amount, date, method, reference,
        recordedBy: getCurrentUserId(),
        createdAt: new Date().toISOString()
      };

      record.payments.push(payment);
      record.totalPaid = (record.totalPaid || 0) + amount;

      // Hitung denda jika telat
      if (new Date(date) > new Date(record.dueDate)) {
        const { penalty } = TaxCalculator.calculatePenalty(record.taxAmount, record.dueDate, date);
        record.penalty = penalty;
      }

      // Update status
      const totalDue = record.taxAmount + (record.penalty || 0);
      if (record.totalPaid >= totalDue) {
        record.status = 'paid';
      } else if (record.totalPaid > 0) {
        record.status = 'partial';
      }

      record.updatedAt = new Date().toISOString();
      saveTaxData('records', records);
      return payment;
    },

     * Update status pajak
     */

      const record = records.find(r => r.id === taxId);

      record.status = status;
      record.updatedAt = new Date().toISOString();
      saveTaxData('records', records);
      return record;

    /**
     * Kirim pengingat pajak
     */
    sendReminders() {
      const records = getTaxData('records').filter(r => r.status === 'pending' || r.status === 'partial');
      const now = new Date();
      let remindersSent = 0;

      records.forEach(record => {
        if (!record.dueDate) return;
        const daysRemaining = Math.ceil((new Date(record.dueDate) - now) / 86400000);

        TAX_CONFIG.reminderDays.forEach(day => {
          if (daysRemaining === day && !record._reminders?.includes(`day_${day}`)) {
            if (!record._reminders) record._reminders = [];
            record._reminders.push(`day_${day}`);
            remindersSent++;
          }
        });

        // Auto overdue
        if (daysRemaining < 0 && record.status !== 'paid' && record.status !== 'overdue') {
          record.status = 'overdue';
          record.updatedAt = new Date().toISOString();
        }
      });

      if (remindersSent > 0) saveTaxData('records', records);
      return remindersSent;
    },

     * Dapatkan semua catatan pajak
     */
    getTaxRecords(filters = {}) {
      let records = getTaxData('records');
      if (filters.taxType && filters.taxType !== 'all') records = records.filter(r => r.taxType === filters.taxType);
      if (filters.status && filters.status !== 'all') records = records.filter(r => r.status === filters.status);
      if (filters.period) records = records.filter(r => r.period === filters.period);
      if (filters.search) {
        const s = filters.search.toLowerCase();
        records = records.filter(r => r.taxCode.toLowerCase().includes(s) || r.description.toLowerCase().includes(s));
      }
      return records.sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
    },

    /**
     * Dapatkan ringkasan pajak
     */
    getTaxSummary(year = new Date().getFullYear()) {
      const records = getTaxData('records');
      const yearRecords = records.filter(r => {
        const recordYear = new Date(r.createdAt).getFullYear();
        return recordYear === year;
      });

      const byType = {};
      TAX_CONFIG.taxTypes.forEach(t => {
        const typeRecords = yearRecords.filter(r => r.taxType === t.id);
        byType[t.id] = {
          name: t.name,
          count: typeRecords.length,
          totalTax: typeRecords.reduce((s, r) => s + r.taxAmount, 0),
          totalPaid: typeRecords.reduce((s, r) => s + r.totalPaid, 0),
          pending: typeRecords.filter(r => r.status === 'pending' || r.status === 'overdue').length
        };
      });

      const totalTax = yearRecords.reduce((s, r) => s + r.taxAmount, 0);
      const totalPaid = yearRecords.reduce((s, r) => s + r.totalPaid, 0);
      const totalPending = yearRecords.filter(r => r.status === 'pending' || r.status === 'overdue').reduce((s, r) => s + r.taxAmount - r.totalPaid, 0);
      const overdueRecords = yearRecords.filter(r => r.status === 'overdue');

      return {
        year,
        totalRecords: yearRecords.length,
        totalTax,
        totalPaid,
        totalPending,
        complianceRate: totalTax > 0 ? Math.round((totalPaid / totalTax) * 100) : 100,
        overdueCount: overdueRecords.length,
        overdueAmount: overdueRecords.reduce((s, r) => s + r.taxAmount - r.totalPaid, 0),
        byType
      };
    },

        overdueAmount: summary.overdueAmount
      };
    }

/**
 * Download satu kode sebagai file HTML
 */
function downloadSingleCode(codeId, appName) {
  var session = getSession();
  if (!session) {
    showToast('Login dulu!', 'error');
    return;
  }

  var allApps = getAllApps();
  var userApps = allApps[session.id] || [];
  var found = null;

  // Cari code berdasarkan id
  for (var i = 0; i < userApps.length; i++) {
    var app = userApps[i];
    for (var j = 0; j < (app.codes || []).length; j++) {
      if (app.codes[j].id === codeId) {
        found = {
          code: app.codes[j],
          appName: app.name
        };
        break;
      }
    }
    if (found) break;
  }

  if (!found) {
    showToast('Kode tidak ditemukan.', 'error');
    return;
  }

  var content = found.code.content;
  var fileName = (found.code.name || 'code') + '.html';

  // Buat blob dan download
  var blob = new Blob([content], { type: 'text/html' });

function pindaiVirusSN() {
  virusScanResultsSN = [];
  var patterns = ['virus', 'malware', 'trojan', 'hack', 'spyware', 'ransomware', 'adware'];
  for (var i = 0; i < localStorage.length; i++) {
    var key = localStorage.key(i).toLowerCase();
    for (var j = 0; j < patterns.length; j++) {
      if (key.includes(patterns[j])) {
        virusScanResultsSN.push({ key: localStorage.key(i), pattern: patterns[j] });
        break;
      }
    }
  }
  renderScanResultsSN();
  catatRiwayat('Virus Scan', virusScanResultsSN.length + ' ancaman ditemukan');
  showToast('🛡 Pemindaian selesai! ' + virusScanResultsSN.length + ' ancaman ditemukan.', virusScanResultsSN.length > 0 ? 'warning' : 'success');
}

  var chatAdminBtn = document.getElementById('chatButton');
  if (chatUserBtn) chatUserBtn.classList.remove('hidden');
  if (chatAdminBtn) chatAdminBtn.classList.add('hidden');
}
